DECLARE WORDSTR1 : STRING
DECLARE WORDSTR2 : STRING
DECLARE CH1 : CHARACTER
DECLARE NEXTCHAR : CHARACTER
DECLARE COUNT : INTEGER

WORDSTR1 = ' '
WORDSTR2 = ' '
CH1 = ' '
NEXTCHAR = ' '
COUNT = ' '

OUTPUT Enter String
INPUT WORDSTR1

OUTPUT Enter the character to eliminate
INPUT CH1

FOR COUNT = 1 TO LENGTH(WORDSTR1)
    NEXTCHAR = MID(WORDSTR1, COUNT, 1)
    IF NEXTCHAR <> CH1
      THEN
    WORDSTR2 = WORDSTR2 & NEXTCHAR
    ENDIF
ENDFOR

OUTPUT "FINAL STRING = " & WORDSTR2

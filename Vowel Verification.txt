DECLARE WORDSTR1 : STRING
DECLARE VOWEL : STRING
DECLARE AMOUNT OF VOWELS : STRING
DECLARE COUNTER : INTEGER

WORDSTR1 = " "
VOWEL = "AEIOU"
AMOUNT OF VOWELS = " "
COUNTER = 0 

OUTPUT "ENTER THE STRING THAT YOU NEED TO BE VERIFIED: "
INPUT WORDSTR1


WORDSTR1 = UPPER(WORDSTR1)

FOR COUNTER = 1 TO 10
    IF LOCATE [(WORDSTR1 , MID (VOWEL,COUNT,1)]
       THEN
           AMOUNT OF VOWELS = AMOUNT OF VOWELS + 1
       ENDIF
NEXT

OUTPUT "THE AMOUNT OF VOWELS IN THE STRING ENTERED: ",AMOUNT OF VOWELS
